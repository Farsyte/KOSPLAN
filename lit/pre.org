pre - precomputed ship data.

* Import all dependencies.

#+begin_src ks
  import("math").
#+end_src

* swet[s] -- wet mass at each stage.

** Approach: add up masses, append to list.

#+begin_src ks
  set swet to list().
  local s is 0.
  until s>stage:number {
    local m is 0.
    for pt in ship:parts
      if pt:decoupledin<s
        set m to m+pt:wetmass.
    swet:add(m).
  set s to s+1.}
  // print "SWET: ["+swet:join(", ")+"]".
#+end_src

* sthr[s] -- precomputed maxthrust for each stage.

#+begin_src ks
  set sthr to list().
  local s is 0.
  until s>stage:number {
    local t is 0.
    for en in ship:engines
      if en:stage>=s and en:decoupledIn<s
      set t to t+en:possiblethrustat(0).
    sthr:add(t).
  set s to s+1.}
  // print "STHR: ["+sthr:join(", ")+"]".
#+end_src

* sisp[s] -- precomputed (G0*Isp) for each stage.

#+begin_src ks
  set sisp to list().
  local s is 0.
  until s>stage:number {
    local i is 0.
    local t is sthr[s].
    if t>0 {
      for en in ship:engines
        if en:stage>=s and en:decoupledIn<s
        set i to i+en:ispat(0)*en:possiblethrustat(0).
    set i to g0*i/t. }
    sisp:add(i).
  set s to s+1.}
  // print "SISP: ["+sisp:join(", ")+"]".
#+end_src
