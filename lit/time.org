time - simple time stuff

* Import Depencencies

#+begin_src ks :results none :tangle ../Archive/time.ks
  import("math").
  import("nv").
#+end_src

* T0 -- universal time of mission launch

Missions are expected to change T0 several times.

- before launch time has been scueduled:
    SET T0 to nvput("t0", inf).
- when launch time is being scheduled:
    SET T0 to nvput("t0", time:seconds+DT).
- when launch actually occurs:
    SET T0 to time:seconds.

#+begin_src ks :results none :tangle ../Archive/time.ks
  set t0 to nvget(1, "t0", inf).
#+end_src

* MET -- return seconds since T0, or -3600 if no T0 set.

Note that if T0 is not yet selected (T0=inf), indicate
that we are in launch-hold at T-1 hour.

#+begin_src ks :results none :tangle ../Archive/time.ks
  local met_h is { return -3600. }.
  local met_f is { return time:seconds - t0. }.
  set met to choose met_f if t0 < inf else met_h.
#+end_src

* T0PUT(dt) -- establish T0 value.

#+begin_src ks :results none :tangle ../Archive/time.ks
  set t0put to { parameter dt.
    set t0 to nvput(1, "t0", TIME:SECONDS+dt).
    set met to choose met_f if t0 < inf else met_h.
    return t0. }.
#+end_src

* ydhms(t) -- format time for printing

Uses TIME:FULL then removes zero elements from the front.

#+begin_src ks :results none :tangle ../Archive/time.ks
  set ydhms to { parameter sec.
    set sec to round(sec).
    if sec<0 return "-"+ydhms(-sec).
    local ts is TIMESTAMP(round(sec)).
    if ts:year<2 and ts:day<2 return ts:CLOCK.
    if ts:year<2 return (ts:day-1)+"d+"+ts:CLOCK.
    return (ts:year-1)+"y "+(ts:day-1)+"d+"+ts:CLOCK.
  }.
#+end_src

* TEE() -- printable MET as "T[+-]###y##h##m#.##s"

#+begin_src ks :results none :tangle ../Archive/time.ks
  set TEE to { local t is met().
    return "T"+(choose "-" if t<0 else "+")+ydhms(abs(t)). }.
#+end_src
